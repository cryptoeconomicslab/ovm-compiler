@library
 def IsTx(tx, token, range, block_number) :=
   Equal(tx, $txAddress)
   and Equal(tx.0, token)
   and IsContained(range, tx.1)
   and IsLessThan(block_number, tx.2)

 @library
 @quantifier("tx.block${b}.range${token},RANGE,${range}")
 def Tx(tx, token, range, b) :=
   IsTx(tx, token, range, b)

def stateUpdate(token, range, block_number, so, subrange) :=
  IsContained(subrange, range)
  and Tx(token, subrange, block_number).any(tx ->
    so(tx)
  )
