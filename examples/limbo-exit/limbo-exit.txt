def LimboExit(prev_su, tx, su) :=
  exit(prev_su)
  or (
    prev_su()
    and IsValidStateTransition(prev_su, tx, su)
    and exit(su))

def IsValidStateTransition(prev_su, tx, su) :=
  eq(tx.adderss, Tx.address)
  and eq(tx.0, prev_su.0)
  and within(tx.1, prev_su.1)
  and eq(tx.2, prev_su.2)
  and eq(tx.3, su.3)
