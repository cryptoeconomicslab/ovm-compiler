def order(maker, c_token, c_amount, min_block_number) := 
  with Bytes() as tx {
    SameRange(tx, self)
    and with SU() as c_su {
      assert(c_su.0, c_token)
      and checkAmount(c_su.1, c_amount)
      and gte(c_su.2, min_block_number)
      and assert(c_su.3, maker)
      and ((withdraw(c_su) and IsValidSignature(tx, c_su.3)) or (!withdraw(c_su) and IsValidSignature(tx, maker)))
    }
  }

def exit_correspondent(c_su, maker) :=
  (exit(c_su) and deposit_exists(c_su.0, c_su.1)) or (
    with Tx(c_su.1) as tx {
      IsValidSignature(maker)
    }
  )
